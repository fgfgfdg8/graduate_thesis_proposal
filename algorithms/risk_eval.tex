\begin{algorithm}[htbp]
\caption{Tool-Chain Risk Evaluation (SeqRiskEval)}
\label{alg:risk_eval}
\begin{algorithmic}[1]
\Require Execution log $\mathcal{L}$, tool registry $\mathcal{T}$, task specification $q$
\Ensure Risk score $R(C)$, contribution vector $\mathbf{contrib}$, risk type $\hat{r}$
\State Parse $\mathcal{L}$ into event sequence $C = \langle e_1, \ldots, e_T \rangle$ using unified log schema
\State \textbf{// Compute six risk indicators}
\State $S_{\text{tool}} \leftarrow \frac{1}{T}\sum_{t=1}^{T} \mathcal{T}[tool_t].\text{sensitivity}$
\State $D_{\text{call}} \leftarrow \max_{t} \text{depth}(e_t)$ \Comment{max delegation depth in chain}
\State $P_{\text{diff}} \leftarrow |Perm(C) \setminus Perm_{req}(q)| \;/\; (|Perm(C) \cup Perm_{req}(q)| + \epsilon)$
\State $I_{\text{irrev}} \leftarrow \frac{1}{T}\sum_{t=1}^{T} irr(e_t)$
\State $E_{\text{expo}} \leftarrow \frac{\sum_{t} \mathbb{I}[o_t \cap Data_s \neq \varnothing] \cdot sink(e_t)}{|Data_s| + \epsilon}$
\State $H_{\text{cov}} \leftarrow \frac{|\{e_t : irr(e_t)=1 \wedge confirm(e_t)=\top\}|}{|\{e_t : irr(e_t)=1\}| + \epsilon}$
\State $\Phi(C) \leftarrow [S_{\text{tool}}, D_{\text{call}}, P_{\text{diff}}, I_{\text{irrev}}, E_{\text{expo}}, H_{\text{cov}}]$
\State \textbf{// Entropy-based adaptive weight calibration}
\For{each indicator $j \in \{1,\ldots,6\}$}
  \State $H_j \leftarrow -\frac{1}{\ln N}\sum_{i=1}^{N} p_{ij} \ln(p_{ij} + \epsilon)$
  \State $\alpha_j \leftarrow (1 - H_j) / \sum_k (1 - H_k)$
\EndFor
\State \textbf{// Compute composite risk score}
\State $R(C) \leftarrow \alpha_s S_{\text{tool}} + \alpha_d D_{\text{call}} + \alpha_p P_{\text{diff}} + \alpha_i I_{\text{irrev}} + \alpha_e E_{\text{expo}} - \alpha_h H_{\text{cov}}$
\State $contrib_j \leftarrow \alpha_j \Phi_j(C) / (\sum_k |\alpha_k \Phi_k(C)| + \epsilon)$ for each $j$
\State \textbf{// Risk type classification by threshold rules}
\If{$P_{\text{diff}} > \theta_p$} $\hat{r} \leftarrow R_1$ \Comment{over-privilege}
\ElsIf{$E_{\text{expo}} > \theta_e$} $\hat{r} \leftarrow R_2$ \Comment{data exfiltration}
\ElsIf{$I_{\text{irrev}} > 0 \wedge H_{\text{cov}} < \theta_h$} $\hat{r} \leftarrow R_3$ \Comment{unconfirmed irreversible}
\Else{} $\hat{r} \leftarrow R_0$ \Comment{normal or other}
\EndIf
\State \Return $R(C)$, $\mathbf{contrib}$, $\hat{r}$
\end{algorithmic}
\end{algorithm}
